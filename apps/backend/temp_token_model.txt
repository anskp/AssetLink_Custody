model Token {
  id                  Int              @id @default(autoincrement())
  userId              Int
  name                String
  symbol              String
  description         String?
  tokenType           TokenType
  assetCategory       String
  templateId          String
  network             TokenNetwork
  selectedNetwork     String?
  blockchain          String
  totalSupply         String?
  decimals            Int?
  maxSupply           String?
  contractAddress     String?
  adminAddress        String?
  minterAddress       String?
  pauserAddress       String?
  vaultId             String
  gasVaultId          String           @default("23")
  tokenDeployer       String?
  status              TokenStatus      @default(CREATING)
  transactionId       String?
  transactionHash     String?
  blockNumber         Int?
  gasUsed             String?
  networkFee          String?
  metadata            Json?
  errorMessage        String?
  createdAt           DateTime         @default(now())
  updatedAt           DateTime         @updatedAt
  deployedAt          DateTime?
  assetId             Int?             @unique
  imageUrl            String?
  burnReason          String?
  burned              Boolean          @default(false)
  burnedAt            DateTime?
  burnedBy            Int?
  frozen              Boolean          @default(false)
  frozenAt            DateTime?
  frozenBy            Int?
  frozenReason        String?
  marketplaceListings Marketplace[]
  tokenOwnerships     TokenOwnership[]
  asset               Asset?           @relation("AssetToToken", fields: [assetId], references: [id])
  user                User             @relation(fields: [userId], references: [id])

  @@index([userId])
  @@index([status])
  @@index([tokenType])
  @@index([contractAddress])
  @@map("tokens")
}
